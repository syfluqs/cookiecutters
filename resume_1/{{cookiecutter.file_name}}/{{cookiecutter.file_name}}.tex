\newcommand{\boxedsectionheading}[1]{
    \vspace{-15pt}
    \Large\color{white}
    \setlength{\fboxsep}{0cm}%already boxed
    \colorbox{black!70}{
        \begin{minipage}{\linewidth}
            \centering
            \vspace*{2pt}%Space before
            \MakeUppercase{\namefont{#1}}
            \vspace*{2pt}%Space after
        \end{minipage}
    }
    \vspace{-20pt}
}

\newlength{\spacebox}

\newcommand{\sepspace}{\vspace*{1em}}		% Vertical space macro

\newcommand{\MyName}[1]{ % Name
    \Huge \hfill \textrm{\MakeUppercase{\namefont{
        \fontsize{35pt}{35pt}\selectfont
        #1
    }}}
    \normalsize
    {\color{black!70}\rule{\textwidth}{1.5pt}}
    \par
    \vspace{-5pt}
}

\newcommand{\ContactDetails}[3]{
    \vspace*{-5pt}
    \begin{center}
        \href{tel:#1}{#1} % phone 
        $\bullet$
        \href{mailto:#2}{#2} % email
        $\bullet$
        \url{#3} % website
    \end{center}
    \vspace*{-10pt}
}
		
\newcommand{\MySlogan}[1]{ % Slogan (optional)
		\large \usefont\hfill \textit{#1}
		\par \normalsize \normalfont}

\newcommand{\NewPart}[1]{
    \section*{{#1}}
}

\newcommand{\PersonalEntry}[2]{
		\noindent\hangindent=2em\hangafter=0 % Indentation
		\parbox{\spacebox}{        % Box to align text
		\textit{#1}}		       % Entry name (birth, address, etc.)
		\hspace{1.5em} #2 \par}    % Entry value

\newcommand{\SkillsEntry}[2]{
    \begin{minipage}{0.3\linewidth}
        \centering
        \textit{#1}
    \end{minipage}
    \hfill
    \begin{minipage}{0.65\linewidth}
        #2
    \end{minipage}
    \vspace{5pt}
}
		
\newcommand{\EducationEntry}[5]{
    \noindent \textbf{#1} \hfill      % Study
    \colorbox{White}{%
        \parbox{5cm}{%
        \hfill\color{Black}
            \textbf{#2}
        }} \par  % Duration
    \noindent \textit{\textbf{#3} - #4} \par        % School
    \noindent\hangindent=2em\hangafter=0 \small #5 % Description
    \normalsize \par
    \vspace{5pt}
}

\newcommand{\WorkEntry}{\EducationEntry}

\newcommand{\ProjectEntry}[4]{

        }} \par  % Duration

    \vspace{5pt}

\newcommand{\PublicationEntry}[5]{

    \colorbox{White}{%

    \noindent [ \url{#5} ]
    \normalsize \par
    \vspace{5pt}
}
